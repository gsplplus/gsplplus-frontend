// src/app/contact/page.tsx
"use client";
import { useState } from "react";
import styles from "./page.module.css";

// 유효성 검사 규칙
const VALIDATION = {
  name: { required: true, minLength: 2, maxLength: 50 },
  email: { required: true, pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/ },
  phone: { required: false, pattern: /^01[0-9]-?[0-9]{3,4}-?[0-9]{4}$/ },
  projectType: { required: true },
  message: { required: true, minLength: 10, maxLength: 2000 },
};

export default function ContactPage() {
  const [formData, setFormData] = useState({
    name: "", email: "", phone: "", projectType: "", message: ""
  });
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [submitResult, setSubmitResult] = useState<"success" | "error" | null>(null);

  // 유효성 검사 함수
  const validate = () => {
    const newErrors: Record<string, string> = {};
    
    if (!formData.name || formData.name.length < 2) {
      newErrors.name = "이름을 2자 이상 입력해주세요.";
    }
    if (!VALIDATION.email.pattern.test(formData.email)) {
      newErrors.email = "올바른 이메일 형식을 입력해주세요.";
    }
    if (formData.phone && !VALIDATION.phone.pattern.test(formData.phone)) {
      newErrors.phone = "올바른 전화번호 형식을 입력해주세요.";
    }
    if (!formData.projectType) {
      newErrors.projectType = "프로젝트 유형을 선택해주세요.";
    }
    if (!formData.message || formData.message.length < 10) {
      newErrors.message = "문의 내용을 10자 이상 입력해주세요.";
    }
    
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!validate()) return;
    
    setIsSubmitting(true);
    try {
      // API 호출 (실제 구현 시 교체)
      await new Promise(resolve => setTimeout(resolve, 1000));
      setSubmitResult("success");
      setFormData({ name: "", email: "", phone: "", projectType: "", message: "" });
    } catch {
      setSubmitResult("error");
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <div className={styles.page}>
      <main className={styles.main}>
        <section className={styles.hero}>
          <p className={styles.eyebrow}>문의하기</p>
          <h1>프로젝트 상담</h1>
          <p className={styles.heroLead}>
            임베디드 개발 관련 문의사항을 남겨주시면<br />
            <strong>1영업일 내</strong> 담당자가 연락드립니다.
          </p>
        </section>

        <div className={styles.content}>
          <form className={styles.form} onSubmit={handleSubmit}>
            <div className={styles.field}>
              <label htmlFor="name">이름 *</label>
              <input
                type="text" id="name" name="name"
                value={formData.name}
                onChange={e => setFormData({...formData, name: e.target.value})}
                placeholder="홍길동"
              />
              {errors.name && <span className={styles.error}>{errors.name}</span>}
            </div>

            <div className={styles.field}>
              <label htmlFor="email">이메일 *</label>
              <input
                type="email" id="email" name="email"
                value={formData.email}
                onChange={e => setFormData({...formData, email: e.target.value})}
                placeholder="example@company.com"
              />
              {errors.email && <span className={styles.error}>{errors.email}</span>}
            </div>

            <div className={styles.field}>
              <label htmlFor="phone">전화번호</label>
              <input
                type="tel" id="phone" name="phone"
                value={formData.phone}
                onChange={e => setFormData({...formData, phone: e.target.value})}
                placeholder="010-1234-5678"
              />
              {errors.phone && <span className={styles.error}>{errors.phone}</span>}
            </div>

            <div className={styles.field}>
              <label htmlFor="projectType">프로젝트 유형 *</label>
              <select
                id="projectType" name="projectType"
                value={formData.projectType}
                onChange={e => setFormData({...formData, projectType: e.target.value})}
              >
                <option value="">선택해주세요</option>
                <option value="hardware">하드웨어 설계/개발</option>
                <option value="software">임베디드 소프트웨어</option>
                <option value="integration">시스템 통합</option>
                <option value="inspection">점검장비/시뮬레이터</option>
                <option value="other">기타 문의</option>
              </select>
              {errors.projectType && <span className={styles.error}>{errors.projectType}</span>}
            </div>

            <div className={styles.field}>
              <label htmlFor="message">문의 내용 *</label>
              <textarea
                id="message" name="message" rows={6}
                value={formData.message}
                onChange={e => setFormData({...formData, message: e.target.value})}
                placeholder="프로젝트 개요, 요구사항, 예상 일정 등을 알려주세요."
              />
              {errors.message && <span className={styles.error}>{errors.message}</span>}
            </div>

            <button type="submit" className={styles.submitBtn} disabled={isSubmitting}>
              {isSubmitting ? "전송 중..." : "문의하기"}
            </button>

            {submitResult === "success" && (
              <p className={styles.success}>문의가 접수되었습니다. 빠른 시일 내 연락드리겠습니다.</p>
            )}
            {submitResult === "error" && (
              <p className={styles.error}>전송에 실패했습니다. 다시 시도해주세요.</p>
            )}
          </form>

          <aside className={styles.sidebar}>
            <div className={styles.infoCard}>
              <h3>연락처</h3>
              <p>📞 042-XXX-XXXX</p>
              <p>📧 info@gspls.kr</p>
            </div>
            <div className={styles.infoCard}>
              <h3>주소</h3>
              <p>대전광역시 유성구<br />배울1로 277, 4동</p>
            </div>
            <div className={styles.infoCard}>
              <h3>응답 안내</h3>
              <p>문의 후 <strong>1영업일 내</strong> 답변</p>
              <p>평일 09:00 ~ 18:00</p>
            </div>
          </aside>
        </div>
      </main>
    </div>
  );
}

  return (
    <>
      <button
        className={styles.hamburger}
        onClick={() => setIsOpen(!isOpen)}
        aria-label="메뉴 열기"
        aria-expanded={isOpen}
      >
        <span className={`${styles.bar} ${isOpen ? styles.open : ""}`} />
        <span className={`${styles.bar} ${isOpen ? styles.open : ""}`} />
        <span className={`${styles.bar} ${isOpen ? styles.open : ""}`} />
      </button>

      {isOpen && (
        <div className={styles.overlay} onClick={() => setIsOpen(false)}>
          <nav className={styles.menu} onClick={(e) => e.stopPropagation()}>
            {navItems.map((item) => (
              <Link
                key={item.href}
                href={item.href}
                className={styles.menuItem}
                onClick={() => setIsOpen(false)}
              >
                {item.label}
              </Link>
            ))}
            <hr className={styles.divider} />
            <Link href="/contact" className={styles.ctaBtn} onClick={() => setIsOpen(false)}>
              문의하기
            </Link>
            <div className={styles.contactInfo}>
              <p>📞 042-XXX-XXXX</p>
              <p>📧 info@gspls.kr</p>
            </div>
          </nav>
        </div>
      )}
    </>
  );
}
